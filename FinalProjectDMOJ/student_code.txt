import java.util.LinkedList;\nimport java.util.Queue;\nimport java.util.Scanner;\n\n/**\n * Problem: State which problem you are tackling\n * @author Jackie Zou\n */\npublic class StudentCode  {\n    private static final int[][] moves = {\n        {-2, -1}, {-1, -2}, {1, -2}, {2, -1}, \n        {2, 1}, {1, 2}, {-1, 2}, {-2, 1}\n    };\n      \n    public static void main(String[] args) {\n        Scanner sc = new Scanner(System.in);\n        \n        // Input starting position\n        //System.out.print("startXY: ");\n        int startX = sc.nextInt();\n        int startY = sc.nextInt();\n        \n        //System.out.print("endXY: ");\n        // Input target position\n        int endX = sc.nextInt();\n        int endY = sc.nextInt();\n        \n         boolean[][] visited = new boolean[8][8];\n\n        // Output the minimum number of moves\n         System.out.println(minKnightMoves(startX, startY, endX, endY));\n        \n    }\n     public static int minKnightMoves(int startX, int startY, int endX, int endY) {\n        // If the starting and ending positions are the same\n        if (startX == endX && startY == endY) {\n            return 0;\n        }\n\n        // Create a queue for BFS\n        Queue<int[]> queue = new LinkedList<>();\n        boolean[][] visited = new boolean[8][8];\n\n        // Add the starting position to the queue\n        queue.offer(new int[]{startX, startY, 0}); // {x, y, moves}\n        visited[startX - 1][startY - 1] = true;\n\n        while (!queue.isEmpty()) {\n            int[] current = queue.poll();\n            int x = current[0];\n            int y = current[1];\n            int movesCount = current[2];\n\n            // Try all possible knight moves\n            for (int[] m : moves) {\n                int newX = x + m[0];\n                int newY = y + m[1];\n\n                // Check if the new position is within bounds and not visited\n                if (newX >= 1 && newX <= 8 && newY >= 1 && newY <= 8 && !visited[newX - 1][newY - 1]) {\n                    // If we reach the target position, return the moves count\n                    if (newX == endX && newY == endY) {\n                        return movesCount + 1;\n                    }\n\n                    // Mark the position as visited and add it to the queue\n                    visited[newX - 1][newY - 1] = true;\n                    queue.offer(new int[]{newX, newY, movesCount + 1});\n                }\n            }\n        }\n\n        // If no solution is found (should not happen in a chessboard)\n        return -1;\n    }\n}\n
